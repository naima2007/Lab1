# Demander le prénom à l'utilisateur
prenom = input("Quel est votre prénom ? ")

# Affichage avec concaténation classique
print("Bonjour " + prenom)

# Affichage avec une f-string (recommandé)
print(f"Ravi de te rencontrer, {prenom} !")

# Affichage de la longueur du texte avec len()
print(f"Longueur du prénom : {len(prenom)} caractères.")# Lab1
